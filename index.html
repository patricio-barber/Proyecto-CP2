#Problema 8

set.seed(123) # establecer semilla para reproducibilidad

# Definir una función para simular una sola vez la generación de números aleatorios uniformes en el intervalo (0,1)
simulacion <- function() {
  suma <- 0
  n <- 0
  while (suma < 1) {
    n <- n + 1
    suma <- suma + runif(1)
  }
  return(n)
}

# Definir un vector de valores de n para simular
n_values <- c(100, 1000, 10000, 100000)

# Realizar n simulaciones para cada valor de n y guardar los resultados en una lista
resultados <- lapply(n_values, function(n) replicate(n, simulacion()))

# Calcular la función de masa de probabilidad aproximada de N para cada valor de n
fmp_aprox <- lapply(resultados, function(x) table(x) / length(x))

# Crear una lista de gráficos para cada función de masa de probabilidad aproximada
graficos <- lapply(seq_along(n_values), function(i) {
  fmp <- fmp_aprox[[i]]
  n <- names(fmp)
  p <- ggplot(data.frame(n, fmp), aes(x = as.numeric(n), y = fmp)) +
    geom_bar(stat = "identity", fill = "steelblue") +
    xlab("Número de sumandos (N)") +
    ylab("Probabilidad") +
    ggtitle(paste("8 n =", n_values[i])) +
    theme(plot.title = element_text(hjust = 0.5))
  return(p)
})

# Combinar todos los gráficos en un solo gráfico utilizando patchwork
p <- graficos[[1]]
for (i in 2:length(graficos)) {
  p <- p + graficos[[i]]
}
